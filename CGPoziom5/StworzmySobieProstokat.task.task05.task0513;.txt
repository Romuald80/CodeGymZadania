/* 
Stwórzmy sobie prostokąt
*/

public class Prostokat {
    //tutaj wpisz swój kod
    private int gora;
    private int lewy;
    private int wysokosc;
    private int szerokosc;

    public void inicjalizuj(int gora, int lewy, int wysokosc, int szerokosc){
        this.gora = gora;
        this.lewy = lewy;
        this.wysokosc = wysokosc;
        this.szerokosc = szerokosc;
    }

    public void inicjalizuj(int gora, int lewy){
        this.gora = gora;
        this.lewy = lewy;
        this. wysokosc = 0;
        this.szerokosc = 0;
    }
    public void inicjalizuj(int gora, int lewy, int szerokosc){
        this.gora = gora;
        this.lewy = lewy;
        this.wysokosc = szerokosc;
        this.szerokosc = szerokosc;
    }

    public void inicjalizuj(Prostokat rectangle){  //tworzenia kopii prostokata
        this.gora = rectangle.gora;
        this.lewy = rectangle.lewy;
        this.szerokosc = rectangle.szerokosc;
        this.wysokosc = rectangle.wysokosc;
    }

    public static void main(String[] args) {

    }
}

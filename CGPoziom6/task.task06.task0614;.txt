package pl.codegym.task.task06.task0614;

import java.util.ArrayList;

/* 
Statyczne koty
*/

public class Kot {
    public static ArrayList<Kot> koty = new ArrayList<>();  // Tworzy obiekt baza danych lista Kot





    public Kot() {
    }

    public static void main(String[] args) {
        for(int i = 0;i <10; i++) koty.add (new Kot()); // za kazdym razem kiedy tworzony jest nowy obiekt(Kot)dodaje go do zmiennej "kot"
        printKoty();					// tworzy 10 Obiektów (Kot)
    }

    public static void printKoty() {// metoda printKoty wyswiertla na ekranie wszystkie 10 obieków
        for(int i = 0; i < koty.size();i++) System.out.println(koty.get(i));
    } // koniec printKot
} // koniec class Kot
